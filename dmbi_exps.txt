import csv

def load_csv(filename):
    with open(filename, 'r') as file:
        reader = csv.reader(file)
        headers = next(reader)
        data = [row for row in reader]
    return headers, data

def class_counts(data):
    counts = {}
    for row in data:
        label = row[-1]
        counts[label] = counts.get(label, 0) + 1
    return counts

def attr_given_class(data, attr_index, attr_value, class_value):
    count = 1  # Laplace smoothing
    total = 0
    unique_vals = set()

    for row in data:
        if row[-1] == class_value:
            total += 1
            unique_vals.add(row[attr_index])
            if row[attr_index] == attr_value:
                count += 1

    return count / (total + len(unique_vals))

def predict_naive_bayes(data, headers, input_row):
    labels = class_counts(data)
    total_rows = len(data)
    probs = {}

    for label in labels:
        prob = labels[label] / total_rows

        for i in range(len(input_row)):
            prob *= attr_given_class(data, i, input_row[i], label)

        probs[label] = prob

    print("Class Probabilities:", probs)
    return max(probs, key=probs.get)

# -------- RUN --------

headers, data = load_csv("dataset.csv")

test_row = ["<=30", "High", "No", "Fair"]

prediction = predict_naive_bayes(data, headers[:-1], test_row)

print("Prediction:", prediction)

import csv

def load_csv(filename):
    with open(filename, 'r') as f:
        reader = csv.reader(f)
        headers = next(reader)
        data = [row for row in reader]
    return headers, data

def conditional_prob(data, headers, conditions, target_value):
    match_count = 0
    target_match = 0

    for row in data:
        match = True
        for attr, val in conditions.items():
            index = headers.index(attr)
            if row[index] != val:
                match = False
                break

        if match:
            match_count += 1
            if row[-1] == target_value:
                target_match += 1

    # Laplace smoothing
    return (target_match + 1) / (match_count + 2)

# -------- RUN --------

headers, data = load_csv("dataset.csv")

conditions = {
    "Student": "Yes",
    "CreditRating": "Fair"
}

p_yes = conditional_prob(data, headers, conditions, "Yes")
p_no = conditional_prob(data, headers, conditions, "No")

print(f"P(Yes | conditions) = {p_yes:.2f}")
print(f"P(No | conditions) = {p_no:.2f}")

prediction = "Yes" if p_yes > p_no else "No"
print("Prediction:", prediction)

import csv
import math

# Load dataset
def load_csv(filename):
    with open(filename, 'r') as file:
        reader = csv.reader(file)
        next(reader)
        data = [row for row in reader]
    return data

# Encode categorical values
def encode(row):
    age_map = {'<=30': 0, '31-40': 1, '>40': 2}
    income_map = {'Low': 0, 'Medium': 1, 'High': 2}

    age = age_map[row[0]]
    income = income_map[row[1]]
    student = 1 if row[2] == 'Yes' else 0
    credit = 1 if row[3] == 'Excellent' else 0
    label = 1 if row[4] == 'Yes' else -1

    return [age, income, student, credit], label

# Train decision stump
def train_stump(X, y, weights):
    n_features = len(X[0])
    best_feature = 0
    best_threshold = 0
    best_polarity = 1
    min_error = float('inf')

    for feature in range(n_features):
        values = set(x[feature] for x in X)

        for thresh in values:
            for polarity in [1, -1]:
                error = 0

                for i in range(len(X)):
                    pred = polarity * (1 if X[i][feature] <= thresh else -1)
                    if pred != y[i]:
                        error += weights[i]

                if error < min_error:
                    min_error = error
                    best_feature = feature
                    best_threshold = thresh
                    best_polarity = polarity

    return best_feature, best_threshold, best_polarity, min_error

# AdaBoost
def adaboost(X, y, T=5):
    n = len(X)
    weights = [1/n] * n
    classifiers = []

    for _ in range(T):
        feat, thresh, polarity, error = train_stump(X, y, weights)

        eps = 1e-10
        alpha = 0.5 * math.log((1 - error + eps) / (error + eps))

        classifiers.append((feat, thresh, polarity, alpha))

        # Update weights
        for i in range(n):
            pred = polarity * (1 if X[i][feat] <= thresh else -1)
            weights[i] *= math.exp(-alpha * y[i] * pred)

        # Normalize
        total = sum(weights)
        weights = [w / total for w in weights]

    return classifiers

# Predict
def predict(x, classifiers):
    final = 0
    for feat, thresh, polarity, alpha in classifiers:
        pred = polarity * (1 if x[feat] <= thresh else -1)
        final += alpha * pred
    return 1 if final > 0 else -1

# -------- RUN --------

data = load_csv("dataset.csv")

X, y = [], []
for row in data:
    features, label = encode(row)
    X.append(features)
    y.append(label)

model = adaboost(X, y, T=5)

# Test input (same format as dataset)
test_row = ['<=30', 'High', 'No', 'Excellent', '?']
x_test, _ = encode(test_row)

result = predict(x_test, model)

print("Prediction:", "Yes" if result == 1 else "No")

import csv
import random

# Load dataset
def load_csv(filename):
    with open(filename, 'r') as file:
        reader = csv.reader(file)
        next(reader)
        return [row for row in reader]

# Encode categorical values
def encode(row):
    age_map = {'<=30': 0, '31-40': 1, '>40': 2}
    income_map = {'Low': 0, 'Medium': 1, 'High': 2}

    age = age_map[row[0]]
    income = income_map[row[1]]
    student = 1 if row[2] == 'Yes' else 0
    credit = 1 if row[3] == 'Excellent' else 0
    label = 1 if row[4] == 'Yes' else -1

    return [age, income, student, credit], label

# Gini impurity
def gini(y):
    total = len(y)
    if total == 0:
        return 0
    p = sum(1 for i in y if i == 1) / total
    return 1 - p*p - (1-p)*(1-p)

# Split dataset
def split(X, y, feature, threshold):
    left_X, left_y, right_X, right_y = [], [], [], []
    for i in range(len(X)):
        if X[i][feature] <= threshold:
            left_X.append(X[i])
            left_y.append(y[i])
        else:
            right_X.append(X[i])
            right_y.append(y[i])
    return left_X, left_y, right_X, right_y

# Build tree
def build_tree(X, y, depth=0, max_depth=5):
    if len(set(y)) == 1 or depth == max_depth:
        return max(set(y), key=y.count)

    best_feature, best_thresh, best_score = 0, 0, float('inf')

    for f in range(len(X[0])):
        values = set(x[f] for x in X)
        for t in values:
            lx, ly, rx, ry = split(X, y, f, t)
            score = (len(ly)*gini(ly) + len(ry)*gini(ry)) / len(X)

            if score < best_score:
                best_feature, best_thresh, best_score = f, t, score
                best_split = (lx, ly, rx, ry)

    lx, ly, rx, ry = best_split

    return ((best_feature, best_thresh),
            build_tree(lx, ly, depth+1),
            build_tree(rx, ry, depth+1))

# Predict tree
def predict_tree(tree, x):
    if isinstance(tree, int):
        return tree
    feature, thresh = tree[0]
    if x[feature] <= thresh:
        return predict_tree(tree[1], x)
    else:
        return predict_tree(tree[2], x)

# Random Forest
def random_forest(X, y, n_trees=5):
    forest = []

    for _ in range(n_trees):
        sample_X, sample_y = [], []
        for _ in range(len(X)):
            i = random.randint(0, len(X)-1)
            sample_X.append(X[i])
            sample_y.append(y[i])

        tree = build_tree(sample_X, sample_y)
        forest.append(tree)

    return forest

# Voting
def predict_forest(forest, x):
    preds = [predict_tree(tree, x) for tree in forest]
    return max(set(preds), key=preds.count)

# -------- RUN --------

data = load_csv("dataset.csv")

X, y = [], []
for row in data:
    features, label = encode(row)
    X.append(features)
    y.append(label)

forest = random_forest(X, y, n_trees=5)

# Test input
test_row = ['<=30', 'High', 'Yes', 'Fair', '?']
x_test, _ = encode(test_row)

result = predict_forest(forest, x_test)

print("Prediction:", "Yes" if result == 1 else "No")

❓ What is bootstrap sampling?

❓ What is voting?
